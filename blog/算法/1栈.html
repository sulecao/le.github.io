# 用数组模拟实现栈

function Stack() {
    let items = [];

    this.push = function (item) {
        items.push(item)
    };
    this.pop = function () {
        return items.pop()
    }
    this.top = function () {
        return items[items.length - 1]
    }
    this.isEmpty = function () {
        return items.length === 0
    }
    this.size = function () {
        return items.length
    }
    this.clear = function () {
        items = []
    }
}

# mins栈

可以获取栈的最小数据的栈

function MinStack() {
    let items = [];
    let mins = []

    this.push = function (item) {
        items.push(item)
        //用mins存储对应的最小值，若顶部min比新的数据大，用新的数据，否则还是以前min。
        if (!mins.length || mins[mins.length - 1] > item) {
            mins.push(item)
        } else {
            mins.push(mins[mins.length - 1])
        }
    };
    this.pop = function () {
        mins.pop()
        return items.pop()
    }
    this.min = function () {
        return mins[mins.length - 1]
    }
}

# 有效的括号 (opens new window)

给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串 s ，判断字符串是否有效。

function isValid(s) {
    if (s.length % 2) {
        return false
    }
    const doc = {
        ')': '(',
        ']': '[',
        '}': '{',
    }
    let stack = []
    for (let i = 0; i < s.length; i++) {
        if ([')', '}', ']'].includes(s[i])) {
            if (doc[s[i]] !== stack.pop()) {
                return false
            }
        } else {
            stack.push(s[i])
        }
    }
    return stack.length == 0
}

# 逆波兰表达式求值 (opens new window)

根据逆波兰表示法 (opens new window)，求表达式的值。如：["2","1","+","3","*"]

function evalRPN(tokens) {
    let stack = []
    for (let i = 0; i < tokens.length; i++) {
        if (['+', '-', '*', '/'].includes(tokens[i])) {
            // 先取出来的值在表达式右边
            let right = stack.pop()
            let left = stack.pop()
            if (tokens[i] == '+') {
                stack.push(left + right)
            } else if (tokens[i] == '-') {
                stack.push(left - right)
            } else if (tokens[i] == '*') {
                stack.push(left * right)
            } else if (tokens[i] == '/') {
                stack.push(parseInt(left / right))
            }
        } else {
            stack.push(parseInt(tokens[i]))
        }
    }
    return stack.pop()
};
